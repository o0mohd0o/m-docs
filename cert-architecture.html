Module Declaration (module.xml)

<?xml version="1.0"?>
<config xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" 
        xsi:noNamespaceSchemaLocation="urn:magento:framework:Module/etc/module.xsd">
    <module name="Vendor_Module" setup_version="1.0.0">
        <sequence>
            <module name="Magento_Catalog"/>
            <module name="Magento_Customer"/>
        </sequence>
    </module>
</config>

Module Registration (registration.php)

<?php
use Magento\Framework\Component\ComponentRegistrar;

ComponentRegistrar::register(
    ComponentRegistrar::MODULE,
    'Vendor_Module',
    __DIR__
);

Dependency Injection

Constructor Injection

<?php
namespace Vendor\Module\Model;

use Magento\Framework\App\ResourceConnection;
use Psr\Log\LoggerInterface;

class MyModel
{
    private $resourceConnection;
    private $logger;

    public function __construct(
        ResourceConnection $resourceConnection,
        LoggerInterface $logger
    ) {
        $this->resourceConnection = $resourceConnection;
        $this->logger = $logger;
    }
}

di.xml Configuration

<?xml version="1.0"?>
<config xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">
    <!-- Preference: Interface to Class mapping -->
    <preference for="Vendor\Module\Api\DataInterface" 
                type="Vendor\Module\Model\Data"/>
    
    <!-- Virtual Type -->
    <virtualType name="CustomLogger" type="Monolog\Logger">
        <arguments>
            <argument name="name" xsi:type="string">customLog</argument>
        </arguments>
    </virtualType>
    
    <!-- Type Configuration -->
    <type name="Vendor\Module\Model\MyModel">
        <arguments>
            <argument name="data" xsi:type="array">
                <item name="key" xsi:type="string">value</item>
            </argument>
        </arguments>
    </type>
</config>

Request Flow

Frontend Request Flow

  1. index.php ‚Üí Bootstrap application
  2. FrontController ‚Üí Route request
  3. Router ‚Üí Match route to controller
  4. Controller ‚Üí Execute action
  5. Layout ‚Üí Load layout XML
  6. Block ‚Üí Process business logic
  7. Template ‚Üí Render HTML
  8. Response ‚Üí Send to browser

Service Contracts

Data Interface

<?php
namespace Vendor\Module\Api\Data;

interface ProductInterface
{
    const SKU = 'sku';
    const NAME = 'name';
    
    /**
     * @return string
     */
    public function getSku();
    
    /**
     * @param string $sku
     * @return $this
     */
    public function setSku($sku);
    
    /**
     * @return string
     */
    public function getName();
    
    /**
     * @param string $name
     * @return $this
     */
    public function setName($name);
}

Repository Interface

<?php
namespace Vendor\Module\Api;

use Vendor\Module\Api\Data\ProductInterface;

interface ProductRepositoryInterface
{
    /**
     * @param int $id
     * @return \Vendor\Module\Api\Data\ProductInterface
     * @throws \Magento\Framework\Exception\NoSuchEntityException
     */
    public function getById($id);
    
    /**
     * @param \Vendor\Module\Api\Data\ProductInterface $product
     * @return \Vendor\Module\Api\Data\ProductInterface
     * @throws \Magento\Framework\Exception\CouldNotSaveException
     */
    public function save(ProductInterface $product);
    
    /**
     * @param \Vendor\Module\Api\Data\ProductInterface $product
     * @return bool
     * @throws \Magento\Framework\Exception\CouldNotDeleteException
     */
    public function delete(ProductInterface $product);
}

Key Concepts

Concept Description Example
Area Logical components (frontend, adminhtml, webapi_rest, etc.) area="frontend"
Scope Configuration level (global, website, store, store view) ScopeConfigInterface
Object Manager Instantiates objects (avoid direct use) ObjectManager::getInstance()
Factory Auto-generated classes to create new objects ProductFactory
Proxy Lazy loading wrapper class Product\Proxy
